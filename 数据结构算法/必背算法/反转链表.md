```Java
public ListNode reverseList(ListNode list) {
    // 使用notProcessed存储暂未反转的链表
    ListNode notProcessed = list.next;
    // 截断链表
    list.next = null;
    // 使用reversed存储已经完成反转的链表
    ListNode reversed = list;

    // 此时达成上述两链表要求的状态
    while(notProcessed != null) {
        // 1. 从notProcessed中取出头结点
        ListNode node1 = notProcessed;
        notProcessed = notProcessed.next;

        // 2. 将上述结点放置到reversed链头
        node1.next = reversed;
        reversed = node1;
    }
    return reversed;
}
```
